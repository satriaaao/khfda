// --- HELPERS ---
const parseIncludesString = (includesStr) => {
  if (!includesStr) return [];
  return includesStr.split(', ').map(part => {
    const match = part.match(/^(\d+)x\s+(.+)$/);
    if (match) return { name: match[2], qty: parseInt(match[1], 10) };
    const matchSuffix = part.match(/(.+)\s+\((\d+)x\)$/);
    if (matchSuffix) return { name: matchSuffix[1], qty: parseInt(matchSuffix[2], 10) };
    return { name: part, qty: 1 };
  }).filter(item => item.name);
};

const generateIncludesString = (includesArr) => {
  return includesArr.map(item => item.qty > 1 ? `${item.name} (${item.qty}x)` : item.name).join(', ');
};

const formatRupiah = (number) => {
  return new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0 }).format(number);
};

                   <div>
                     <label className="block text-[10px] font-bold text-slate-500 uppercase tracking-wider mb-1">PPN (%)</label>
                     <input type="number" min="0" value={orderForm.tax} onChange={e=>setOrderForm({...orderForm, tax: e.target.value})} className="w-full border border-slate-300 rounded-lg p-2.5 text-sm outline-none focus:border-blue-500" placeholder="0"/>
                   </div>
                   <div>
                     <label className="block text-[10px] font-bold text-slate-500 uppercase tracking-wider mb-1">DP / Uang Muka (Rp)</label>
                     <input type="number" min="0" value={orderForm.paidAmount} onChange={e=>setOrderForm({...orderForm, paidAmount: e.target.value})} className="w-full border border-slate-300 rounded-lg p-2.5 text-sm outline-none focus:border-blue-500" placeholder="0"/>
                   </div>
                </div>
              </div>
            </div>

            {/* Footer Modal (Sticky Bottom di HP) */}
            <div className="p-4 sm:p-6 border-t border-slate-200 bg-white sm:bg-slate-50 flex flex-col sm:flex-row justify-between items-center gap-4 sm:rounded-b-2xl shrink-0 shadow-[0_-4px_6px_-1px_rgba(0,0,0,0.05)] sm:shadow-none">
              <div className="text-sm font-semibold text-slate-600 w-full text-center sm:text-left border-b border-slate-100 sm:border-0 pb-3 sm:pb-0">
                Durasi Sewa: <span className="text-blue-600 px-2 py-0.5 bg-blue-50 rounded-md">{orderForm.totalDays} Hari</span>
              </div>
              <div className="flex flex-col sm:flex-row gap-3 w-full sm:w-auto">
                <button type="button" onClick={() => setIsOrderModalOpen(false)} className="w-full sm:w-auto px-6 py-3 sm:py-2.5 border border-slate-300 rounded-xl text-sm font-semibold text-slate-600 hover:bg-slate-50 bg-white transition-colors order-2 sm:order-1">Batal</button>
                <button type="button" onClick={handleSaveOrder} className="w-full sm:w-auto px-6 py-3 sm:py-2.5 bg-blue-600 text-white rounded-xl text-sm font-semibold shadow-md hover:bg-blue-700 transition-colors order-1 sm:order-2">Simpan Input Jalan</button>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
